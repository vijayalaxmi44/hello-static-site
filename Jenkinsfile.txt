pipeline {
    agent any

    environment {
        DEPLOY_DIR = '/tmp/static-website' // Deployment folder
    }

    stages {
        stage('Clone Code') {
            steps {
                git 'https://github.com/your-username/hello-static-site.git'
            }
        }

        stage('Build') {
            steps {
                echo 'Nothing to build for static files'
            }
        }

        stage('Archive') {
            steps {
                archiveArtifacts artifacts: '*.html, *.css'
            }
        }

        stage('Deploy') {
            steps {
                sh '''
                mkdir -p $DEPLOY_DIR
                cp index.html styles.css $DEPLOY_DIR
                echo "Static site deployed to $DEPLOY_DIR"
                '''
            }
        }
    }

    post {
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed.'
        }
    }
}
